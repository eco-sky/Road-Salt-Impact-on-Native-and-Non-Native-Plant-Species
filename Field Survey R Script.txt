# ----10. Removing Outliers Percentage Data----

sdintroduced<- sd(EVSC_enviro_factors$total_introduced)
sdintroduced

clean_EVSC_enviro_factors<-EVSC_enviro_factors[
  EVSC_enviro_factors$total_cover<0.8 &
  EVSC_enviro_factors$total_invasive<0.8 &
    EVSC_enviro_factors$total_introduced<2*sdintroduced &
    EVSC_enviro_factors$total_native<0.8,]
